# clubactivity
 
# Table 생성(MySQL)

-- club.`member` definition

CREATE TABLE `member` (
  `MEMBER_ID` varchar(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '회원 아이디',
  `MEMBER_PASSWORD` varchar(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '회원 비밀번호',
  `MEMBER_NAME` text CHARACTER SET utf8 COLLATE utf8_general_ci COMMENT '회원이름',
  `MEMBER_PHONE_NUMBER` text CHARACTER SET utf8 COLLATE utf8_general_ci COMMENT '회원전화번호',
  `MEMBER_REGISTRATION_DATE` date DEFAULT NULL COMMENT '회원가입일자',
  `MEMBER_NUMBER` int(11) NOT NULL AUTO_INCREMENT,
  `MEMBER_WEBSITE` varchar(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci DEFAULT NULL COMMENT '회원 사이트',
  `MEMBER_STATUS` tinyint(4) NOT NULL DEFAULT '0' COMMENT '온라인 상태',
  `MEMBER_LEVEL_CODE` int(11) NOT NULL DEFAULT '0' COMMENT '회원등급',
  `IMAGE_NUMBER` int(11) DEFAULT NULL COMMENT '이미지',
  PRIMARY KEY (`MEMBER_NUMBER`),
  KEY `member_FK_1` (`MEMBER_STATUS`),
  KEY `member_FK_2` (`MEMBER_LEVEL_CODE`),
  CONSTRAINT `member_FK_1` FOREIGN KEY (`MEMBER_STATUS`) REFERENCES `memberonline` (`MEMBER_STATUS`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `member_FK_2` FOREIGN KEY (`MEMBER_LEVEL_CODE`) REFERENCES `memberlevel` (`MEMBER_LEVEL_CODE`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=29 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

-- club.memberlevel definition

CREATE TABLE `memberlevel` (
  `MEMBER_LEVEL_CODE` int(11) NOT NULL DEFAULT '0' COMMENT '회원등급',
  `MEMBER_LEVEL_DESCRIPTION` varchar(100) NOT NULL COMMENT '회원상태',
  PRIMARY KEY (`MEMBER_LEVEL_CODE`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

-- club.memberonline definition

CREATE TABLE `memberonline` (
  `MEMBER_STATUS` tinyint(4) NOT NULL DEFAULT '0' COMMENT '온라인 상태',
  `MEMBER_STATUS_DESCRIPTION` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`MEMBER_STATUS`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

-- club.messagecommand definition

CREATE TABLE `messagecommand` (
  `MEMBER_NUMBER` int(11) NOT NULL COMMENT '회원번호',
  `TITLE` varchar(64) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT '제목',
  `MESSAGE` varchar(512) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT '내용',
  `MESSAGE_NUMBER` int(11) NOT NULL AUTO_INCREMENT COMMENT '메시지 번호',
  `READ` tinyint(4) DEFAULT '0' COMMENT '읽었는지 확인',
  `FROM_MEMBER_NUMBER` int(11) DEFAULT NULL COMMENT '발송자',
  PRIMARY KEY (`MESSAGE_NUMBER`),
  KEY `messagecommand_FK` (`MEMBER_NUMBER`),
  CONSTRAINT `messagecommand_FK` FOREIGN KEY (`MEMBER_NUMBER`) REFERENCES `member` (`MEMBER_NUMBER`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=24 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

-- club.image definition

CREATE TABLE `image` (
  `MEMBER_NUMBER` int(11) NOT NULL COMMENT '회원 번호',
  `IMAGE_PATH` varchar(100) NOT NULL COMMENT '이미지',
  `IMAGE_NUMBER` int(11) NOT NULL AUTO_INCREMENT COMMENT 'PK',
  PRIMARY KEY (`IMAGE_NUMBER`),
  KEY `image_FK` (`MEMBER_NUMBER`),
  CONSTRAINT `image_FK` FOREIGN KEY (`MEMBER_NUMBER`) REFERENCES `member` (`MEMBER_NUMBER`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=25 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
